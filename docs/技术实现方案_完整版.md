## 一、项目概述

### 1.1 项目目标
开发一个高性能自动化视频裁剪工具，能够根据参考素材片段，从多个原始视频文件中快速精确匹配、裁剪并拼接，生成与参考素材完全一致的新视频。

### 1.2 核心价值
- **高效性**：采用多级匹配算法，大幅提升处理速度
- **准确性**：音频指纹+视频帧混合匹配，确保精确匹配
- **可扩展性**：模块化设计，支持多种匹配策略
- **易用性**：提供命令行和API两种使用方式

## 三、核心算法实现

### 3.1 多级匹配算法

#### 3.1.1 第一级：音频指纹快速定位（粗匹配）

**实现原理**：
1. 从参考素材和源视频中提取音频指纹
2. 使用滑动窗口匹配算法快速定位候选区间
3. 计算音频相似度，筛选高置信度候选

**技术细节**：
```python
def extract_audio_fingerprint(video_path: Path) -> np.ndarray:
    """
    提取音频指纹（频谱特征）

    步骤：
    1. 使用FFmpeg提取音频流（降采样至8kHz）
    2. 分段音频（1秒一段，50%重叠）
    3. 对每段进行FFT变换获取频谱
    4. 分桶降维（32个频段）
    5. 返回指纹数组
    """
    # 降采样加速处理
    audio_sample_rate = 8000
    segment_size = audio_sample_rate  # 1秒一段
    overlap = 0.5  # 50%重叠

    # 分桶降维
    buckets = 32
    fingerprint_size = buckets  # 每个指纹32维

    return fingerprints
```

**性能优化**：
- 音频降采样至8kHz，减少计算量
- 使用FFT加速频谱计算
- 分桶降维减少特征维度
- 滑动窗口优化搜索效率

#### 3.1.2 第二级：关键帧特征匹配（精确定位）

**实现原理**：
1. 在候选区间内提取关键帧
2. 计算帧特征（直方图、pHash、模板）
3. 与参考素材的关键帧比对
4. 确定精确起始点

**技术细节**：
```python
def extract_frame_features(frame: np.ndarray) -> dict:
    """
    提取帧特征

    特征包括：
    1. 颜色直方图（32维）
    2. 感知哈希（16维）
    3. 模板特征（256维）
    """
    # 颜色直方图
    hist = cv2.calcHist([frame], [0,1,2], None, [8,8,8], [0,256,0,256,0,256])
    hist = cv2.normalize(hist, hist).flatten()

    # 感知哈希
    gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
    resized = cv2.resize(gray, (16,16))
    dct = cv2.dct(np.float32(resized))
    phash = (dct[:8,:8].flatten() > np.mean(dct[:8,:8])).astype(int)

    # 模板特征
    template = cv2.resize(gray, (64,64)).flatten()[:256]

    return {
        'histogram': hist,
        'phash': phash,
        'template': template
    }

def calculate_similarity(feat1: dict, feat2: dict) -> float:
    """
    计算特征相似度

    综合三种特征：
    1. 直方图相关性（40%权重）
    2. 感知哈希汉明距离（30%权重）
    3. 模板余弦相似度（30%权重）
    """
    hist_sim = cv2.compareHist(
        feat1['histogram'], feat2['histogram'],
        cv2.HISTCMP_CORREL
    )

    phash_sim = 1 - (np.sum(feat1['phash'] != feat2['phash']) / 16.0)

    template_sim = np.dot(
        feat1['template'], feat2['template']
    ) / (
        np.linalg.norm(feat1['template']) * 
        np.linalg.norm(feat2['template']) + 1e-10
    )

    return 0.4 * max(0, hist_sim) + 0.3 * phash_sim + 0.3 * max(0, template_sim)
```

**性能优化**：
- 只处理关键帧（I帧）减少计算量
- 使用低分辨率提取特征
- 并行提取多帧特征
- 缓存已提取特征

#### 3.1.3 第三级：连续帧验证（完整性确认）

**实现原理**：
1. 从匹配起始点开始连续比对后续帧
2. 验证内容连续性和相似度
3. 确保找到完整片段

**技术细节**：
```python
def verify_continuity(
    target_frames: List[np.ndarray],
    source_frames: List[np.ndarray],
    start_idx: int,
    verify_frames: int = 10
) -> bool:
    """
    验证连续性

    步骤：
    1. 从起始点开始连续比对N帧
    2. 计算每帧的相似度
    3. 统计高相似度帧的比例
    4. 判断是否通过验证
    """
    high_similarity_count = 0
    threshold = 0.7  # 连续帧相似度阈值

    for i in range(verify_frames):
        if start_idx + i >= len(source_frames):
            break

        sim = calculate_similarity(
            target_frames[i],
            source_frames[start_idx + i]
        )

        if sim > threshold:
            high_similarity_count += 1

    # 要求80%的帧满足相似度要求
    return (high_similarity_count / verify_frames) > 0.8
```

```python
class ParallelProcessor:
    """并发处理器"""

    def __init__(self, max_workers: int = 4):
        self.max_workers = max_workers
        self.executor = ProcessPoolExecutor(max_workers=max_workers)

    def process_videos_parallel(
        self,
        videos: List[Path],
        process_func: Callable
    ) -> List[Result]:
        """并行处理多个视频"""
        futures = []
        for video in videos:
            future = self.executor.submit(process_func, video)
            futures.append(future)

        results = []
        for future in as_completed(futures):
            result = future.result()
            results.append(result)

        return results
```

#### 3.2.2 视频分片并发

```python
def process_video_chunks(
    video_path: Path,
    chunk_size: int = 600  # 10分钟一片
) -> List[Result]:
    """
    分片处理大视频

    步骤：
    1. 获取视频总时长
    2. 按chunk_size分片
    3. 并行处理各分片
    4. 合并处理结果
    """
    duration = get_video_duration(video_path)
    chunks = []

    # 分片
    for start in range(0, int(duration), chunk_size):
        end = min(start + chunk_size, int(duration))
        chunks.append((start, end))

    # 并行处理
    with ThreadPoolExecutor(max_workers=4) as executor:
        futures = [
            executor.submit(
                process_chunk,
                video_path,
                start,
                end
            )
            for start, end in chunks
        ]

        results = [f.result() for f in futures]

    # 合并结果
    return merge_results(results)
```

### 3.3 智能优化特性

#### 3.3.1 动态阈值调整

```python
class DynamicThresholdAdjuster:
    """动态阈值调整器"""

    def __init__(self, initial_threshold: float = 0.85):
        self.threshold = initial_threshold
        self.confidence_history = []

    def adjust_threshold(self, confidence: float) -> float:
        """
        根据匹配置信度调整阈值

        策略：
        1. 记录最近的置信度
        2. 计算平均置信度
        3. 根据平均置信度调整阈值
        """
        self.confidence_history.append(confidence)

        # 只保留最近20次记录
        if len(self.confidence_history) > 20:
            self.confidence_history.pop(0)

        avg_confidence = np.mean(self.confidence_history)

        # 动态调整
        if avg_confidence > 0.9:
            # 高置信度，提高阈值
            self.threshold = min(0.95, self.threshold + 0.02)
        elif avg_confidence < 0.7:
            # 低置信度，降低阈值
            self.threshold = max(0.75, self.threshold - 0.02)

        return self.threshold
```

```python
class FeatureCache:
    """特征缓存"""

    def __init__(self, cache_dir: Path):
        self.cache_dir = cache_dir
        self.cache_dir.mkdir(parents=True, exist_ok=True)
        self.index_file = cache_dir / "cache_index.json"
        self.index = self._load_index()

    def get_cached_features(self, video_path: Path) -> Optional[dict]:
        """获取缓存的特征"""
        video_hash = self._get_video_hash(video_path)

        if video_hash in self.index:
            cache_path = self.cache_dir / f"{video_hash}.pkl"
            if cache_path.exists():
                with open(cache_path, 'rb') as f:
                    return pickle.load(f)

        return None

    def cache_features(self, video_path: Path, features: dict):
        """缓存特征"""
        video_hash = self._get_video_hash(video_path)
        cache_path = self.cache_dir / f"{video_hash}.pkl"

        with open(cache_path, 'wb') as f:
            pickle.dump(features, f)

        self.index[video_hash] = {
            'video_path': str(video_path),
            'cache_time': time.time(),
            'features': list(features.keys())
        }

        self._save_index()

    def _get_video_hash(self, video_path: Path) -> str:
        """计算视频文件哈希"""
        with open(video_path, 'rb') as f:
            return hashlib.md5(f.read()).hexdigest()
```

```python
        self.config = self.DEFAULT_CONFIG.copy()

        if config_path and config_path.exists():
            user_config = self._load_config(config_path)
            self._merge_config(user_config)

    def _load_config(self, config_path: Path) -> dict:
        """加载配置文件"""
        with open(config_path, 'r', encoding='utf-8') as f:
            return yaml.safe_load(f)

    def _merge_config(self, user_config: dict):
        """合并用户配置"""
        def deep_merge(base, update):
            for key, value in update.items():
                if key in base and isinstance(base[key], dict) and isinstance(value, dict):
                    deep_merge(base[key], value)
                else:
                    base[key] = value

        deep_merge(self.config, user_config)

    def get(self, key_path: str, default=None):
        """获取配置值"""
        keys = key_path.split('.')
        value = self.config

        for key in keys:
            if isinstance(value, dict) and key in value:
                value = value[key]
            else:
                return default

        return value

    def set(self, key_path: str, value):
        """设置配置值"""
        keys = key_path.split('.')
        config = self.config

        for key in keys[:-1]:
            if key not in config:
                config[key] = {}
            config = config[key]

        config[keys[-1]] = value
```

## 五、错误处理与容错

### 5.1 匹配失败分级处理

```python
class MatchFailureHandler:
    """匹配失败处理器"""

    def __init__(self, config: dict):
        self.config = config
        self.current_level = 1
        self.max_level = 3

    def handle_failure(self, error: Exception, context: dict) -> Optional[Result]:
        """
        处理匹配失败

        Level 1: 尝试降低相似度阈值重试
        Level 2: 切换到备选匹配算法重试
        Level 3: 手动介入模式（标记待人工确认）
        """
        if self.current_level == 1:
            return self._retry_with_lower_threshold(error, context)
        elif self.current_level == 2:
            return self._retry_with_alternative_algorithm(error, context)
        else:
            return self._mark_for_manual_review(error, context)

    def _retry_with_lower_threshold(self, error: Exception, context: dict) -> Optional[Result]:
        """降低阈值重试"""
        threshold = context.get('threshold', 0.85)
        new_threshold = threshold * 0.9  # 降低10%

        if new_threshold >= 0.5:  # 最低阈值0.5
            context['threshold'] = new_threshold
            self.current_level += 1
            return None  # 返回None表示继续重试

        return self._mark_for_manual_review(error, context)

    def _retry_with_alternative_algorithm(self, error: Exception, context: dict) -> Optional[Result]:
        """使用备选算法重试"""
        strategy = context.get('strategy', 'hybrid')

        if strategy == 'hybrid':
            context['strategy'] = 'video_only'
        elif strategy == 'video_only':
            context['strategy'] = 'audio_first'
        else:
            return self._mark_for_manual_review(error, context)

        self.current_level += 1
        return None

    def _mark_for_manual_review(self, error: Exception, context: dict) -> dict:
        """标记为需要人工确认"""
        return {
            'status': 'failed',
            'error': str(error),
            'requires_manual_review': True,
            'context': context
        }
```

```python
class CheckpointManager:
    """断点续传管理器"""

    def __init__(self, checkpoint_dir: Path):
        self.checkpoint_dir = checkpoint_dir
        self.checkpoint_dir.mkdir(parents=True, exist_ok=True)

    def save_checkpoint(self, video_path: Path, progress: dict):
        """保存检查点"""
        video_hash = self._get_video_hash(video_path)
        checkpoint_file = self.checkpoint_dir / f"{video_hash}.json"

        checkpoint_data = {
            'video_path': str(video_path),
            'video_hash': video_hash,
            'timestamp': time.time(),
            'progress': progress
        }

        with open(checkpoint_file, 'w', encoding='utf-8') as f:
            json.dump(checkpoint_data, f, indent=2, ensure_ascii=False)

    def load_checkpoint(self, video_path: Path) -> Optional[dict]:
        """加载检查点"""
        video_hash = self._get_video_hash(video_path)
        checkpoint_file = self.checkpoint_dir / f"{video_hash}.json"

        if checkpoint_file.exists():
            with open(checkpoint_file, 'r', encoding='utf-8') as f:
                return json.load(f)

        return None

    def clear_checkpoint(self, video_path: Path):
        """清除检查点"""
        video_hash = self._get_video_hash(video_path)
        checkpoint_file = self.checkpoint_dir / f"{video_hash}.json"

        if checkpoint_file.exists():
            checkpoint_file.unlink()

    def _get_video_hash(self, video_path: Path) -> str:
        """计算视频文件哈希"""
        with open(video_path, 'rb') as f:
            return hashlib.md5(f.read()).hexdigest()
```

```python
        self.total_steps = total_steps

🔄 {stage_name}")

        # 计算预计剩余时间
        elapsed = time.time() - self.stage_start_time
        if self.completed_steps > 0:
            time_per_step = elapsed / self.completed_steps
            remaining_steps = self.total_steps - self.completed_steps
            eta = time_per_step * remaining_steps

   进度: {progress:.1f}% ({self.completed_steps}/{self.total_steps}) "
                  f"ETA: {eta:.1f}秒", end='', flush=True)

    def complete_stage(self):
        """完成当前阶段"""

        self.stage_times[self.current_stage] = elapsed

        print(f"
   完成! 耗时: {elapsed:.1f}秒")
```

```python
class PerformanceProfiler:
    """性能分析器"""

    def __init__(self):
        self.metrics = {
            'timing': {},
            'memory': {},
            'cpu': {}
        }
        self.start_times = {}

    def start_timer(self, name: str):
        """开始计时"""
        self.start_times[name] = time.time()

    def stop_timer(self, name: str):
        """停止计时"""
        if name in self.start_times:
            elapsed = time.time() - self.start_times[name]
            self.metrics['timing'][name] = elapsed
            del self.start_times[name]

    def record_memory(self, name: str):
        """记录内存使用"""
        process = psutil.Process()
        memory_info = process.memory_info()
        self.metrics['memory'][name] = {
            'rss_mb': memory_info.rss / 1024 / 1024,
            'vms_mb': memory_info.vms / 1024 / 1024
        }

    def record_cpu(self, name: str):
        """记录CPU使用"""
        process = psutil.Process()
        self.metrics['cpu'][name] = process.cpu_percent()

    def analyze(self) -> dict:
        """分析性能数据"""
        analysis = {
            'timing_analysis': {},
            'memory_analysis': {},
            'recommendations': []
        }

        # 分析耗时
        if self.metrics['timing']:
            total_time = sum(self.metrics['timing'].values())
            for name, elapsed in self.metrics['timing'].items():
                percentage = elapsed / total_time * 100
                analysis['timing_analysis'][name] = {
                    'elapsed': elapsed,
                    'percentage': percentage
                }

                # 识别瓶颈
                if percentage > 30:
                    analysis['recommendations'].append(
                        f"'{name}' 耗时占比 {percentage:.1f}%，建议优化"
                    )

        # 分析内存
        if self.metrics['memory']:
            max_rss = max(m['rss_mb'] for m in self.metrics['memory'].values())
            if max_rss > 2000:  # 超过2GB
                analysis['recommendations'].append(
                    f"内存使用峰值 {max_rss:.1f}MB，建议优化内存使用"
                )

        return analysis

    def print_report(self):
        """打印性能报告"""
        analysis = self.analyze()

        print("
" + "="*60)
        print("📈 性能分析报告")
        print("="*60)

        print("
⏱️ 耗时分析:")
        for name, data in analysis['timing_analysis'].items():
            print(f"  {name}:")
            print(f"    耗时: {data['elapsed']:.1f}秒")
            print(f"    占比: {data['percentage']:.1f}%")

        print("
💡 优化建议:")
        for rec in analysis['recommendations']:
            print(f"  - {rec}")
```
